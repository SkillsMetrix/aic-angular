var module = angular.module('app', []) module.service('ContactService', function () { var uid = 1; var contacts = [] // logic to show the data this.loadUsers = function () { return contacts } // logic to save the data this.save = function (contact) { if (contact.id == null) { contact.id = uid ++; contacts.push(contact); } else { for (i in contacts) { if (contacts[i].id == contact.id) { contacts[i] = contact } } } } }) module.controller('ContactController', function ($scope, ContactService) { $scope.contacts = ContactService.loadUsers() $scope.saveContact=function(){ ContactService.save($scope.newcontact) } })
